Regional
-------------
Regional_id  PK  INT
Regional_name VARCHAR(100)


country_code
-------------
Country_id pk INT
Regional_id INT FK >- regional.regional_id
English_name VARCHAR(100)
Alpha-2_code VARCHAR(100)
Alpha-3_code VARCHAR(100)
Numeric_code INT
ISO_3166-2  VARCHAR(100)
Land_Area INT

yearly
-------------
yearly_id PK INT
past_year INT
